Расположение ./markup/layouts

Директрия в которой находяться основные файлы лейоутов на основе которых будут составленны итоговые *.html файлы.

Лейауты содержат основную стркутуру разметки.


Основные *.ejs файлы из {@tutorial folder-views} используйт лейауты для расширения при помощи метода <% layout('../layouts/main') %>

Пример лейаута из ./markup/layouts/main.ejs


<!DOCTYPE html>
<html lang="ru-RU" dir="ltr" class="no-js">
	<head>
		<%- partial('../views/widgets/head') %>
		<%- partial('../views/widgets/criticals') %>
	</head>
	<body>
		<div class="pageWrapper">
			<% if (_activeFileData.hasSeoText) { %>
				<main id="seoText" class="pageSeoText">
					<div class="pageSize"><%- partial(_activeFileData.seoTextContent, {}) %></div>
				</main>
				<!-- end pageSeoText -->
			<% } %>

			<header class="pageHeader">
				<%- partial('../views/widgets/header', {}) %>
			</header>
			<!-- end pageHeader -->


			<div class="pageContainer">
				<%- body %>
			</div>
			<!-- end pageContainer -->

			<% if (_activeFileData.hasSeoText) { %>
				<div id="seoClone"></div>
			<% } %>


			<footer class="pageFooter">
				<%- partial('../views/widgets/footer', {}) %>
			</footer>
			<!-- end pageFooter -->

		</div>
		<!-- end pageWrapper -->


		<!-- внешние css файлы -->
		<link rel="stylesheet" property="stylesheet" href="css/style.css">
		<link rel="stylesheet" property="stylesheet" href="css/vendor/plugins.css">


		<!-- внешние js файлы -->
		<script src="js/vendor/modernizr.js"></script>
		<script src="js/vendor/jquery-2.2.4.min.js"></script>
		<script src="js/vendor/plugins.js"></script>
		<script src="js/init.js"></script>

	</body>
</html>
