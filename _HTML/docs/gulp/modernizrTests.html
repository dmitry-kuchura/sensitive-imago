modernizrTests

Набор пользовательских тестов для modernizr.js. Смотри workwith-modernizr

test chrome

Определение браузера chrome

Исходный код:


Modernizr.addTest(
	'chrome',
	function() {
		var ua = navigator.userAgent.toLowerCase();
		var chrome = !!window.chrome && ua.match(/Opera|OPR\//i) === null;
		var edje = ua.indexOf(" edge/") > 0;
		return chrome && !edje;
	}
);

test retina

Определение retina дисплеев

Исходный код:


Modernizr.addTest(
	'retina',
	function() {
		var dpr = window.devicePixelRatio || (window.screen.deviceXDPI / window.screen.logicalXDPI) || 1;
		var flag = dpr > 1;
		return !!flag;
	}
);

test webkit

Определение webkit браузеров

Исходный код:


Modernizr.addTest(
	'webkit',
	function() {
		var ua = window.navigator.userAgent.toLowerCase();
		var webkit = 'WebkitAppearance' in docElement.style;
		var edge = !(ua.indexOf(" edge/") > 0);
		return webkit && edge;
	}
);
